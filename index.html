Xodus

Java Embedded Transactional Key-Value Storage

Get Started

Environment env = Environments.newInstance("mydatabase", EnvironmentConfig.DEFAULT);

Store store = env.computeInTransaction(new TransactionalComputable<Store>() {
    public Store compute(Transaction txn) {
        return env.openStore("Users", StoreConfiguration.WITHOUT_DUPLICATES, txn);
    }
});

env.executeInTransaction(new TransactionalExecutable() {
    public void execute(Transaction txn) {
        store.put(txn, StringBinding.stringToEntry("lvo"),
                       StringBinding.stringToEntry("jetbrains"));
    }
});

env.executeInTransaction(new TransactionalExecutable() {
    public void execute(Transaction txn) {
        System.out.println(StringBinding.entryToString(
                    store.get(txn, StringBinding.stringToEntry("lvo"))));
    }
});

env.close();
